<script>


const common = [

"Find someone who printed a calibration cube",

"Find someone who prints with PLA",

"Find someone attending AMUG for the first time",

"Find someone who has printed a Benchy",

"Find someone who uses support structures"

];


const rare = [

"Find someone who prints carbon fiber nylon",

"Find someone who runs multiple printers",

"Find someone who has designed lattice structures",

"Find someone who prints functional hinges",

"Find someone who uses HP MJF technology"

];


const legendary = [

"Find someone who has printed titanium",

"Find someone running a print farm",

"Find someone who has printed aerospace parts",

"Find someone who runs industrial additive systems",

"Find someone who has printed medical devices"

];


function randomQuest(list){

return list[Math.floor(Math.random()*list.length)];

}


function generateQuests(){


const c=randomQuest(common);

const r=randomQuest(rare);

const l=randomQuest(legendary);


document.getElementById("quests").innerHTML=`


<div class="quest">

<h2>COMMON QUEST</h2>

<p>${c}</p>

<button class="printbtn" onclick="printQuest('COMMON', '${c}')">Print</button>

</div>


<div class="quest">

<h2>RARE QUEST</h2>

<p>${r}</p>

<button class="printbtn" onclick="printQuest('RARE', '${r}')">Print</button>

</div>


<div class="quest">

<h2>LEGENDARY QUEST</h2>

<p>${l}</p>

<button class="printbtn" onclick="printQuest('LEGENDARY', '${l}')">Print</button>

</div>


`;

}


function printQuest(type,quest){


const ticket = `

======================

   SIDE QUEST NPC

======================


${type} QUEST


${quest}


Escort them to the

SAVE POINT NPC


Reward: ${type} TOKEN


AMUG 2026

======================

`;


const win = window.open('', '', 'width=300,height=400');

win.document.write('<pre>'+ticket+'</pre>');

win.print();


}
